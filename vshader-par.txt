#version 330

//Uniform variables
uniform mat4 P;
uniform mat4 V;


//Attributes
in vec3 vertex; //vertex coordinates in model space
in vec2 vertexUV;
in vec3 normal; //vertex normal (ignored for now)
in mat4 M;


out vec2 UV;
out vec3 tvertex;
out float dist;

void main(void) {

	gl_Position=P*V*M*vec4(vertex,1);
	UV = vertexUV;
	tvertex = normalize((V*M*vec4(vertex,1)).xyz)-0.5;
	//tvertex = vertex -0.5;

	dist = distance(vec2(0,0), tvertex.xy );
}
